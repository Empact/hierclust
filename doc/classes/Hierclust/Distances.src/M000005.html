    def nearest
      @nearest ||= begin
        key = nearest_pair[0]
        a_id, b_id = key.split ','
        a = @items.select {|item| item.object_id.to_s == a_id}.first
        b = @items.select {|item| item.object_id.to_s == b_id}.first
        [a, b]
      end
    end
